Model:
public class TodosModel : PageModel
{
    private readonly IProjectService _projectService;
    private readonly ITodoService _todoService;

    public ProjectBasicsDto Project { get; set; }
    public List<TodoDto> Todos { get; set; }

    public TodosModel(IProjectService projectService, ITodoService todoService)
    {
        _projectService = projectService;
        _todoService = todoService;
    }

    public async Task<IActionResult> OnGetAsync(int projectId, bool showAll = false)
    {
        Project = await _projectService.GetBasics(projectId);
        Todos = await _todoService.GetTodos(projectId, showAll);

        return Page();
    }

    public async Task<IActionResult> OnPostDeleteTodoAsync(int id)
    {
        await _todoService.DeleteTodo(id);
        return new JsonResult(new { success = true });
    }

    public async Task<IActionResult> OnPostDeletePostAsync(int id)
    {
        await _todoService.DeletePost(id);
        return new JsonResult(new { success = true });
    }

    public async Task<IActionResult> OnPostToggleCompleteAsync(int id)
    {
        await _todoService.ToggleComplete(id);
        return new JsonResult(new { success = true });
    }

    public async Task<IActionResult> OnPostAddOrEditTodoAsync(TodoDto dto)
    {
        await _todoService.AddOrEdit(dto);
        return new JsonResult(new { success = true });
    }
}

javascript:
<script>
    const reloadTodos = () => {
        const projectId = $("#ProjectId").val();
        const showAll = $("#switchShowAll").is(":checked");

        $("#todosContainer").load(`/Projects/Todos/${projectId}?showAll=${showAll} #todosContainer>*`);
    };

    $("#switchShowAll").on("change", reloadTodos);

    $(document).on("click", ".delete-todo", function () {
        const id = $(this).data("id");
        $("#modal-confirm-delete-todo").data("id", id).modal("show");
    });

    $("#modal-confirm-delete-todo .btn-confirm").on("click", function () {
        const id = $("#modal-confirm-delete-todo").data("id");

        $.post("?handler=DeleteTodo", { id }, () => reloadTodos());
    });

    $(document).on("click", ".delete-post", function () {
        const id = $(this).data("id");
        $("#modal-confirm-delete-post").data("id", id).modal("show");
    });

    $("#modal-confirm-delete-post .btn-confirm").on("click", function () {
        const id = $("#modal-confirm-delete-post").data("id");

        $.post("?handler=DeletePost", { id }, () => reloadTodos());
    });

    $(document).on("change", ".toggle-complete", function () {
        const id = $(this).data("id");
        $.post("?handler=ToggleComplete", { id }, () => reloadTodos());
    });

    $("#btnAddTodo").on("click", function () {
        $("#todoId").val("");
        $("#todoTitle").val("");
        $("#todoBody").val("");
        $("#modal-add-edit-todo").modal("show");
    });

    $(document).on("click", ".edit-todo", function () {
        const id = $(this).data("id");

        $.getJSON(`/api/todos/${id}`, function (data) {
            $("#todoId").val(data.id);
            $("#todoTitle").val(data.title);
            $("#todoBody").val(data.body);
            $("#modal-add-edit-todo").modal("show");
        });
    });

    $("#btnSaveTodo").on("click", function () {
        const dto = {
            Id: $("#todoId").val(),
            ProjectId: $("#ProjectId").val(),
            Title: $("#todoTitle").val(),
            Body: $("#todoBody").val()
        };

        $.post("?handler=AddOrEditTodo", dto, () => {
            $("#modal-add-edit-todo").modal("hide");
            reloadTodos();
        });
    });
</script>
