$(document).on('change', '.form-check-input', function () {
    const isCompleted = $(this).is(':checked');
    const posId = $(this).data('pos-id');

    $.ajax({
        url: '/YourController/UpdateCompletion',
        type: 'POST',
        data: {
            id: posId,
            isCompleted: isCompleted
        },
        success: function (response) {
            console.log('Zaktualizowano status:', response);
        },
        error: function () {
            console.error('Błąd podczas aktualizacji statusu');
        }
    });
});

[HttpPost]
public IActionResult UpdateCompletion(int id, bool isCompleted)
{
    var pos = _context.Positions.FirstOrDefault(x => x.Id == id);
    if (pos == null)
        return NotFound();

    pos.IsCompleted = isCompleted;
    _context.SaveChanges();

    return Json(new { success = true });
}
