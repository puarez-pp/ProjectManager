public class GetCriticalPathQueryHandler 
    : IRequestHandler<GetCriticalPathQuery, List<ScheduleTaskDto>>
{
    private readonly IScheduleRepository _scheduleRepository;
    private readonly ICriticalPathService _criticalPathService;

    public GetCriticalPathQueryHandler(
        IScheduleRepository scheduleRepository,
        ICriticalPathService criticalPathService)
    {
        _scheduleRepository = scheduleRepository;
        _criticalPathService = criticalPathService;
    }

    public async Task<List<ScheduleTaskDto>> Handle(
        GetCriticalPathQuery request, 
        CancellationToken cancellationToken)
    {
        var schedule = await _scheduleRepository.GetByIdAsync(request.ScheduleId);

        var tasks = schedule.Stages
            .SelectMany(s => s.Tasks)
            .ToList();

        var criticalPath = _criticalPathService.CalculateCriticalPath(tasks);

        return criticalPath
            .Select(t => new ScheduleTaskDto(t))
            .ToList();
    }
}
